INSTALLATION
$ npm install @solscanner/sdk --save

Quick Start

import { SolScanner } from '@solscanner/sdk';

// Initialize with your API Key
const scanner = new SolScanner('sk_live_...');

async function analyzeToken() {
  try {
    const report = await scanner.audit.token('So111...');
    
    if (report.riskScore < 50) {
      console.log('Token is SAFE');
    } else {
      console.warn('High Risk Detected:', report.vectors);
    }
  } catch (err) {
    console.error(err);
  }
}

analyzeToken();
